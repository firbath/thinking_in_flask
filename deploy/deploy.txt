source #VENV_DIR#/bin/activate
#退出环境
deactivate

pip install -r deploy/requirements.txt -i https://mirrors.aliyun.com/pypi/simple

python src/manager.py db init
python src/manager.py db migrate
python src/manager.py db upgrade
